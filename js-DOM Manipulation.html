        this.renderTasks();
    }

        this.list.innerHTML = '';
        
        const fragment = document.createDocumentFragment();
        
        this.tasks.forEach(task => {

            li.className = `task ${task.completed ? 'completed' : ''}`;
            li.dataset.id = task.id;
            
            li.innerHTML = `
                ${task.text}
                

                    
                    
                

            `;
            
            fragment.appendChild(li);
        });

    }

    saveTasks() {

    }
